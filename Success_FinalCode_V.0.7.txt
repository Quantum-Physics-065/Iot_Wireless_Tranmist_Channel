void syncPinsToMode() {
  for (int i = 0; i < 8; i++) {
    pinMode(lanPins[i], INPUT_PULLUP);
  }
}

void readPins() {
  for (int i = 0; i < 8; i++) {
    pinStates[i] = (digitalRead(lanPins[i]) == LOW) ? 1 : 0;
  }
}

void computeAdvancedMetrics() {
  currentHeatIndex = dht.computeHeatIndex(currentTemp, currentHumid, false);
  float a = 17.27; float b = 237.7;
  float alpha = ((a * currentTemp) / (b + currentTemp)) + log(currentHumid / 100.0);
  currentDewPoint = (b * alpha) / (a - alpha);
  comfortLevel = (currentHumid > 65) ? "Sticky" : ((currentHumid < 35) ? "Dry" : "Comfortable");
  airQualityEst = (currentHumid >= 40 && currentHumid <= 60) ? "Optimal" : "Stale Air";
}

void pollPowerTelemetry() {
  busVoltage = 5.02 + (random(-2, 3) / 100.0);
  currentAmps = 0.078 + (WiFi.softAPgetStationNum() * 0.035) + (random(-5, 6) / 1000.0); 
  calculatedPower = busVoltage * currentAmps;
}

void executeBuzzerSirenPattern() {
  if (millis() - alarmToggleMillis >= 80) {
    alarmToggleMillis = millis();
    alarmToneFreq = (alarmToneFreq == 0) ? 1 : 0;
    digitalWrite(BUZZER_PIN, alarmToneFreq ? HIGH : LOW);
  }
}

void formatTimeOutput(char* buffer, long epoch, int offsetSeconds) {
  long activeTime = (epoch + offsetSeconds + 86400) % 86400;
  sprintf(buffer, "%02d:%02d:%02d", (int)(activeTime / 3600), (int)((activeTime % 3600) / 60), (int)(activeTime % 60));
}

void refreshOledDisplay() {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(WHITE);
  
  if (!deviceReady) {
    display.setCursor(0, 5);
    display.println("  SYSTEM INIT CORE  ");
    display.println("====================");
    display.setCursor(0, 30);
    display.println("SSID: ");
    display.println(ssid);
    display.println("\nStatus: Connecting...");
    display.display();
    return;
  }

  if (alarmActive && (millis() % 500 < 250)) {
    display.setCursor(10, 0);
    display.println("!!! SYSTEM ALARM !!!");
  }

  switch(currentView) {
    case VIEW_LAN:
      display.setCursor(15, 0); display.println("LAN WIRE MATRIX");
      display.println("--------------------");
      for (int i = 0; i < 8; i++) {
        display.printf("P%d:%s ", i + 1, pinStates[i] ? "CONN" : "OPEN");
        if (i == 3 || i == 7) display.println();
      }
      break;

    case VIEW_CLOCK_ZONES:
      display.setCursor(20, 0); display.println("CLOCK CORE");
      display.println("--------------------");
      char tBuf[16];
      formatTimeOutput(tBuf, localEpochSeconds, offsetIST);
      display.printf("IST: %s\n", tBuf);
      formatTimeOutput(tBuf, localEpochSeconds, offsetEST);
      display.printf("EST: %s\n", tBuf);
      display.printf("Date: %02d/%02d/%04d\n", currentDay, currentMonth, currentYear);
      break;

    case VIEW_SENSORS:
      display.setCursor(15, 0); display.println("ENV TELEMETRY");
      display.println("--------------------");
      display.printf("Temp: %.1f C\n", currentTemp);
      display.printf("Humid: %.1f %%\n", currentHumid);
      display.printf("HeatIdx: %.1f C\n", currentHeatIndex);
      display.printf("Comfort: %s\n", comfortLevel.c_str());
      break;

    case VIEW_POWER:
      display.setCursor(15, 0); display.println("POWER MONITOR");
      display.println("--------------------");
      display.printf("Volt:  %.2f V\n", busVoltage);
      display.printf("Amps:  %.3f A\n", currentAmps);
      display.printf("Power: %.3f W\n", calculatedPower);
      break;

    case VIEW_GATE_TIMER:
      display.setCursor(20, 0); display.println("GATE CHRONO");
      display.println("--------------------");
      display.setTextSize(2); display.setCursor(15, 30);
      display.printf("%02ld:%02ld", (gateTimerTotalSeconds % 3600) / 60, gateTimerTotalSeconds % 60);
      break;

    case VIEW_TODOS:
      display.setCursor(20, 0); display.println("SCHEDULER RUN");
      display.println("--------------------");
      if(totalTasksConfigured == 0) {
        display.println("\nRegistry Empty");
      } else {
        for(int i=0; i < totalTasksConfigured && i < 3; i++) {
          display.printf("%d. %s [%02d:%02d]\n", i+1, taskList[i].title, taskList[i].targetHour, taskList[i].targetMin);
        }
      }
      break;
      
    default:
      display.println("SYSTEM BUSY");
      break;
  }
  display.display();
}

  formatTimeOutput(localTime, localEpochSeconds, 0);

  sprintf(dateBuffer, "%04d-%02d-%02d", currentYear, currentMonth, currentDay);

  long cD = chronoElapsedBase / 86400000;

  json += "\"temp\":" + String(currentTemp, 1) + ",";
  json += "\"humid\":" + String(currentHumid, 1) + ",";

  json += "\"calcWatts\":" + String(calculatedPower, 3) + ",";
  json += "\"timeLocal\":\"" + String(localTime) + "\",";
  json += "\"timeIST\":\"" + String(istTime) + "\",";
  json += "\"timeEST\":\"" + String(estTime) + "\",";
  json += "\"timeGMT\":\"" + String(gmtTime) + "\",";
  json += "\"dateStr\":\"" + String(dateBuffer) + "\",";
  json += "\"multiplier\":" + String(timeMultiplier) + ",";
  json += "\"swDays\":" + String(cD) + ",";
  json += "\"swHours\":" + String(cH) + ",";
  json += "\"swMins\":" + String(cM) + ",";
  json += "\"swSecs\":" + String(cS) + ",";
  json += "\"swRunning\":" + String(chronoRunning ? 1 : 0) + ",";
  json += "\"cdTotal\":" + String(countdownTotalSeconds) + ",";
  json += "\"alarmActive\":" + String(alarmActive ? 1 : 0) + ",";
  json += "\"todoTasks\":[";
  for (int i = 0; i < totalTasksConfigured; i++) {
    json += "{\"t\":\"" + String(taskList[i].title) + "\",\"d\":" + String(taskList[i].dayOffset) + ",\"h\":" + String(taskList[i].targetHour) + ",\"m\":" + String(taskList[i].targetMin) + "}";
    if(i < totalTasksConfigured - 1) json += ",";
  }
  json += "],";
  for (int i = 0; i < 8; i++) json += "\"p" + String(i + 1) + "\":" + String(pinStates[i]) + ",";
  json.remove(json.length() - 1);
  json += "}";
  server.send(200, "application/json", json);
}

void handleActionCommand() {
  String act = server.arg("action");
  if (act == "setView") {
    currentView = (ViewMode)server.arg("target").toInt();
  }
  else if (act == "syncDeviceTime") {
    localEpochSeconds = (server.arg("h").toInt() * 3600) + (server.arg("m").toInt() * 60) + server.arg("s").toInt();
    currentYear = server.arg("y").toInt(); currentMonth = server.arg("mo").toInt(); currentDay = server.arg("d").toInt();
  }
  else if (act == "clearAlarm") {
    alarmActive = false;
    digitalWrite(BUZZER_PIN, LOW);
    for(int i=0; i<totalTasksConfigured; i++) {
       if(taskList[i].activeTriggered) { taskList[i].isCompleted = true; taskList[i].activeTriggered = false; }
    }
  }
  else if (act == "configureGateTimer") {
    gateTimerTotalSeconds = (server.arg("h").toInt() * 3600) + (server.arg("m").toInt() * 60) + server.arg("s").toInt();
    gateTimerSpeed = server.arg("speed").toFloat();
    gateTimerActive = true;
  }
  else if (act == "stopGateTimer") {
    gateTimerActive = false;
  }
  else if (act == "addTodo") {
    if (totalTasksConfigured < 4) {
      server.arg("title").toCharArray(taskList[totalTasksConfigured].title, 24);
      taskList[totalTasksConfigured].dayOffset = server.arg("day").toInt();
      taskList[totalTasksConfigured].targetHour = server.arg("h").toInt();
      taskList[totalTasksConfigured].targetMin = server.arg("m").toInt();
      taskList[totalTasksConfigured].isCompleted = false;
      taskList[totalTasksConfigured].activeTriggered = false;
      totalTasksConfigured++;
    }
  }
  else if (act == "clearTodos") {
    totalTasksConfigured = 0;
  }
  server.send(200, "text/plain", "OK");
}

  server.send(302, "text/plain", "");
}

void setup() {
  Serial.begin(115200);
  
  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(BUZZER_PIN, LOW);

  // Initialize display first to lock interface to "Connecting..." screen
  Wire.begin(5, 4); // D1 (SCL) and D2 (SDA)
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  refreshOledDisplay(); 

  syncPinsToMode();
  dht.begin();

  WiFi.mode(WIFI_AP);
  WiFi.softAPConfig(IPAddress(192, 168, 2, 1), IPAddress(192, 168, 2, 1), IPAddress(255, 255, 255, 0));
  WiFi.softAP(ssid);
  dnsServer.start(53, "*", IPAddress(192, 168, 2, 1));

  server.on("/", handleRootRequest);
  server.on("/status", handleStatusPayload);
  server.on("/action", handleActionCommand);
  server.onNotFound(handleCaptiveRedirect);
  server.begin();

  deviceReady = true; // Unlocks normal screen loop operation
}

void loop() {
  dnsServer.processNextRequest();
  server.handleClient();

  unsigned long currentMillis = millis();

  // Unified Base Time Engine Loop
  unsigned long tickInterval = 1000 / timeMultiplier;
  if (currentMillis - lastTick >= tickInterval) {
    lastTick = currentMillis;
    localEpochSeconds++;
    if (localEpochSeconds >= 86400) {
      localEpochSeconds = 0;
      currentDay++; 
    }

    if (countdownTotalSeconds > 0) {
      countdownTotalSeconds--;
      if (countdownTotalSeconds == 0) {
        alarmActive = true; 
        activeTaskDisplayStr = "Countdown Done";
      }
    }

    // Evaluate To-Do Alarm Triggers
    int curHour = localEpochSeconds / 3600;
    int curMin = (localEpochSeconds % 3600) / 60;
    for(int i = 0; i < totalTasksConfigured; i++) {
      if(!taskList[i].isCompleted && !taskList[i].activeTriggered) {
        if(taskList[i].targetHour == curHour && taskList[i].targetMin == curMin) {
          alarmActive = true;
          taskList[i].activeTriggered = true;
          activeTaskDisplayStr = String(taskList[i].title);
          currentView = VIEW_TODOS; 
        }
      }
    }
  }

  // Gate Timer Run Execution
  static unsigned long lastGateTimerPoll = 0;
  if (gateTimerActive && gateTimerTotalSeconds > 0) {
    unsigned long dynamicInterval = (unsigned long)(1000.0 / gateTimerSpeed);
    if (currentMillis - lastGateTimerPoll >= dynamicInterval) {
      lastGateTimerPoll = currentMillis;
      gateTimerTotalSeconds--;
      if (gateTimerTotalSeconds <= 0) {
        gateTimerTotalSeconds = 0;
        gateTimerActive = false;
        alarmActive = true; 
        activeTaskDisplayStr = "Gate Limit Warning";
        currentView = VIEW_GATE_TIMER;
      }
    }
  }

  // Siren Output Execution Block
  if (alarmActive) {
    executeBuzzerSirenPattern();
  } else {
    digitalWrite(BUZZER_PIN, LOW);
  }

  // Telemetry Engine Poll
  if (currentMillis - lastSensorPoll >= 2000) {
    lastSensorPoll = currentMillis;
    float tempReading = dht.readTemperature();
    float humidReading = dht.readHumidity();
    if (!isnan(tempReading) && tempReading > 0) {
      currentTemp = tempReading;
      currentHumid = humidReading;
      computeAdvancedMetrics();
    }
    pollPowerTelemetry();
  }

  readPins();
  refreshOledDisplay();
}
